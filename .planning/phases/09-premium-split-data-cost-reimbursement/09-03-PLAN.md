    /// @title HedgeDataMeter — per-position INCREMENTAL hedge-data cost meter (v1 stubbed write).
    /// @notice Implements the FROZEN ICostMeter.cost read seam via a per-position cumulative accumulator
    ///         in native decimals. The mutualized $199 lives in the vault, NEVER here (P5).
    contract HedgeDataMeter is IHedgeMeter {
        address public immutable keeper;                 // v1: the test/keeper stub; HEDGE-01: live keeper
        mapping(address => uint256) private _c0;         // cumulative incremental, token0 native dp {fx: —}

        error NotKeeper();
        constructor(address keeper_) { keeper = keeper_; }

        /// @inheritdoc ICostMeter
        function cost(address p) external view returns (uint256 cost0, uint256 cost1) {
