       ```yaml
         sync:
           name: Sync abrigo content
           runs-on: ubuntu-latest

           steps:
             - name: Checkout frontend
               uses: actions/checkout@v4

             - name: Setup pnpm
               uses: pnpm/action-setup@v4

             - name: Setup Node
               uses: actions/setup-node@v4
               with:
                 node-version-file: '.nvmrc'
                 cache: 'pnpm'

             - name: Copy abrigo scratch + docs markdown into content/research/
               run: |
                 mkdir -p content/research
                 # Only copy *.md and *.mdx files; preserve nested directory shape if needed
                 rsync -av --include='*/' --include='*.md' --include='*.mdx' --exclude='*' \
                   _abrigo/scratch/ content/research/scratch/ || true
                 rsync -av --include='*/' --include='*.md' --include='*.mdx' --exclude='*' \
                   _abrigo/docs/ content/research/docs/ || true
                 echo "Synced files:"
                 find content/research/scratch content/research/docs -type f 2>/dev/null || true

             - name: Install dependencies
               run: pnpm install --frozen-lockfile

             - name: Validate via Velite
               run: pnpm velite build

             - name: Detect changes
               id: detect
               run: |
                 if [[ -n "$(git status --porcelain content/)" ]]; then
                   echo "has_changes=true" >> "$GITHUB_OUTPUT"
                 else
                   echo "has_changes=false" >> "$GITHUB_OUTPUT"
                 fi

             - name: Create Pull Request
               if: steps.detect.outputs.has_changes == 'true'
               uses: peter-evans/create-pull-request@v6
               with:
                 token: ${{ secrets.GITHUB_TOKEN }}
                 commit-message: "content: sync abrigo iteration and research content"
                 title: "content: sync iteration and research content from abrigo-analytics"
                 body: |
                   Automated content sync from `wvs-finance/abrigo-analytics`.

                   Triggered by: `${{ github.event_name }}` (ref=`${{ github.event.client_payload.ref || inputs.abrigo_ref || 'main' }}`)

                   Merge gate: regular CI (impeccable, lint, typecheck, vitest, playwright, lighthouse, a11y).
                 branch: auto/sync-abrigo-content
                 base: main
                 delete-branch: true
                 add-paths: |
                   content/
       ```

       Key design points:
       - The rsync `--include='*/'` + `--include='*.md' --include='*.mdx' --exclude='*'` pattern only copies markdown files but preserves directory structure (so notebooks/ remains in subdirectories).
       - The detect step prevents empty PRs.
       - `delete-branch: true` on create-pull-request cleans up the auto/* branch after merge.
       - The PR triggers the existing ci.yml workflow because it pushes to a new branch (which `on: push:` or `pull_request:` will catch).
