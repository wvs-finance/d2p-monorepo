       function walk(dir: string): string[] {
         const out: string[] = []
         for (const name of readdirSync(dir)) {
           const p = join(dir, name)
           if (statSync(p).isDirectory()) out.push(...walk(p))
           else if (/\.(ts|tsx)$/.test(p)) out.push(p)

         return out
