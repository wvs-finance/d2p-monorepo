    ```solidity
    /// @notice Decompose a premium into its three economic slices with an EXACT-SUM remainder sink.
    /// @param premium   the gross premium  {value: premium,    unit: USD-proxy (USDC 6dp), fx: —}
    /// @param phiData   the MUTUALIZED data slice (vault-sourced; NOT computed here, P5)
    ///                  {value: phiData, unit: USD-proxy (USDC 6dp), fx: —}
    /// @param piWeightBps  pi_panoptic share of (premium − phiData), in bps [0, 10_000]
    /// @return piPanoptic  {value, unit: USD-proxy 6dp, fx: —}
    /// @return muLP        REMAINDER sink {value, unit: USD-proxy 6dp, fx: —}
    /// @return phiData_    echoed phiData {value, unit: USD-proxy 6dp, fx: —}
    function split(uint256 premium, uint256 phiData, uint256 piWeightBps)
        public pure returns (uint256 piPanoptic, uint256 muLP, uint256 phiData_)
    {
        require(phiData <= premium, "phiData>premium");
        require(piWeightBps <= 10_000, "bps>1e4");
        uint256 hedgeable = premium - phiData;
        piPanoptic = Math.mulDiv(hedgeable, piWeightBps, 10_000);
        muLP = hedgeable - piPanoptic;          // REMAINDER SINK ⇒ Σ == premium EXACTLY
        phiData_ = phiData;
        // invariant: piPanoptic + muLP + phiData == premium  (no dust leak, RESEARCH Pattern 1)
    }
    ```
